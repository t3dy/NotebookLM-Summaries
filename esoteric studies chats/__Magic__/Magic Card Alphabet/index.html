import requests
import string

# Function to fetch card data from Scryfall API
def fetch_cards(query, page=1):
    url = f"https://api.scryfall.com/cards/search"
    params = {"q": query, "page": page}
    response = requests.get(url, params=params)
    if response.status_code == 200:
        return response.json()
    else:
        return None

# Function to check if text contains all letters of the alphabet
def contains_all_letters(text):
    alphabet = set(string.ascii_lowercase)
    text_letters = set(text.lower())
    return alphabet.issubset(text_letters)

# Search the Scryfall database
def search_cards():
    query = "++"  # Use Scryfall query syntax to search all cards
    page = 1
    while True:
        data = fetch_cards(query, page)
        if not data or "data" not in data:
            break
        for card in data["data"]:
            card_text = f"{card.get('name', '')} {card.get('type_line', '')} {card.get('oracle_text', '')} {card.get('flavor_text', '')}"
            if contains_all_letters(card_text):
                print(f"Found a card with all 26 letters: {card['name']}")
                print(f"Details: {card['scryfall_uri']}")
                return
        if not data.get("has_more", False):
            break
        page += 1
    print("No card found containing all 26 letters.")

# Run the search
search_cards()

    response = requests.get(url, params=params)
    if response.status_code == 200:
        return response.json()
    else:
        print(f"Error fetching data: {response.status_code}")
        return None

# Function to check if text contains all letters of the alphabet

    alphabet = set(string.ascii_lowercase)
    text_letters = set(text.lower())
    return alphabet.issubset(text_letters)

# Search the Scryfall database

    query = "oracle:*"  # Use Scryfall query syntax to search all cards
    page = 1
    found = False
    while True:
        data = fetch_cards(query, page)
        if not data or "data" not in data:
            break
        for card in data["data"]:
            # Combine all relevant text fields
            card_text = f"{card.get('name', '')} {card.get('type_line', '')} {card.get('oracle_text', '')} {card.get('flavor_text', '')}"

                found = True
                return
        
        # Check if there are more pages of results
        if not data.get("has_more", False):
            break
        page += 1
    
    if not found:
        print("No card found containing all 26 letters.")

import requests
import string

# Function to fetch card data from Scryfall API
def fetch_cards(query, page=1):
    url = f"https://api.scryfall.com/cards/search"
    params = {"q": query, "page": page}
    response = requests.get(url, params=params)
    if response.status_code == 200:
        return response.json()
    elif response.status_code == 404:
        print("Error: Query returned no results or exceeded valid pages.")
        return None
    else:
        print(f"Error fetching data: {response.status_code}")
        return None

# Function to check if text contains all letters of the alphabet
def contains_all_letters(text):
    alphabet = set(string.ascii_lowercase)
    text_letters = set(text.lower())
    return alphabet.issubset(text_letters)

# Search the Scryfall database
def search_cards():
    query = "oracle:*"  # Search all cards with Oracle text
    page = 1
    found = False
    while True:
        data = fetch_cards(query, page)
        if not data or "data" not in data:
            break
        for card in data["data"]:
            # Combine all relevant text fields
            card_text = f"{card.get('name', '')} {card.get('type_line', '')} {card.get('oracle_text', '')} {card.get('flavor_text', '')}"
            
            # Debugging: Print card name and text being analyzed
            print(f"Checking card: {card['name']}")
            print(f"Text: {card_text}")
            
            # Check if this card contains all 26 letters
            if contains_all_letters(card_text):
                print(f"Found a card with all 26 letters: {card['name']}")
                print(f"Details: {card['scryfall_uri']}")
                found = True
                return
        
        # Check if there are more pages of results
        if not data.get("has_more", False):
            break
        page += 1
    
    if not found:
        print("No card found containing all 26 letters.")

# Run the search
if __name__ == "__main__":
    search_cards()
