                }

                Item item = new Item(data.ID, data.Name, data.Description, itemType);
                item.Effects = data.Effects;
                item.GrimoireReferences = data.GrimoireReferences;

                // Link the item to the grimoires
                foreach (string grimoireName in item.GrimoireReferences)

                    Grimoire grimoire = grimoires.Find(g => g.Name == grimoireName);
                    if (grimoire != null)
